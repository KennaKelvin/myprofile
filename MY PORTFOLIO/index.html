Python Backend Sketch

A minimal Flask REST API — the kind of backend that powers your study notes app or any data-driven project.

api/app.py  —  Study Notes REST API (Flask + SQLite) 
# ── Study Notes REST API ──────────────────────────────────────
# Author : Ikenna Iroegbu
# Stack  : Python · Flask · SQLite · REST
# Run    : python app.py  →  http://localhost:5000

from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

app = Flask(__name__)
app.config["SQLALCHEMY_DATABASE_URI"] = "sqlite:///notes.db"
db = SQLAlchemy(app)

# ── Model ──────────────────────────────────────────────────────
class Note(db.Model):
    id      = db.Column(db.Integer, primary_key=True)
    date    = db.Column(db.String(20),  nullable=False)
    topic   = db.Column(db.String(120), nullable=False)
    note    = db.Column(db.Text,         nullable=False)
    points  = db.Column(db.Text,         nullable=True)
    created = db.Column(db.DateTime,     default=datetime.utcnow)

    def to_dict(self):
        return {
            "id": self.id, "date": self.date,
            "topic": self.topic, "note": self.note,
            "points": self.points,
            "created": self.created.isoformat()
        }

# ── Routes ─────────────────────────────────────────────────────
@app.route("/notes", methods=["GET"])
def get_notes():
    notes = Note.query.order_by(Note.created.desc()).all()
    return jsonify([n.to_dict() for n in notes])

@app.route("/notes", methods=["POST"])
def create_note():
    data = request.get_json()
    note = Note(
        date=data["date"], topic=data["topic"],
        note=data["note"],  points=data.get("points", "")
    )
    db.session.add(note)
    db.session.commit()
    return jsonify(note.to_dict()), 201

@app.route("/notes/<int:note_id>", methods=["PUT"])
def update_note(note_id):
    note = Note.query.get_or_404(note_id)
    data = request.get_json()
    note.note   = data.get("note",   note.note)
    note.points = data.get("points", note.points)
    db.session.commit()
    return jsonify(note.to_dict())

@app.route("/notes/<int:note_id>", methods=["DELETE"])
def delete_note(note_id):
    note = Note.query.get_or_404(note_id)
    db.session.delete(note)
    db.session.commit()
    return jsonify({"message": "Note deleted"}), 200

if __name__ == "__main__":
    with app.app_context():
        db.create_all()   # create tables on first run
    app.run(debug=True)
What this does: A complete REST API for your study notes app — GET all notes, POST new note, PUT to edit, DELETE to remove. Drop your frontend's localStorage calls and point them here instead. Install with: pip install flask flask-sqlalchemy
